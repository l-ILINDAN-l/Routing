UPDATE table_road_1
SET length = subquery.distance
FROM (
SELECT
r.start_id,
r.end_id,
(
6371 * acos(
cos(radians(p1.latitude*111)) * cos(radians(p2.latitude*111)) *
cos(radians(p2.longitude*111) - radians(p1.longitude*111)) +
sin(radians(p1.latitude*111)) * sin(radians(p2.latitude*111))
)
) AS distance
FROM
table_road_1 r
JOIN
table_points1 p1 ON r.start_id = p1.point_id
JOIN
table_points1 p2 ON r.end_id = p2.point_id
) AS subquery
WHERE
table_road_1.start_id = subquery.start_id
AND table_road_1.end_id = subquery.end_id;
